1	/**
2	 * Sink behavior dispatch. Given a Sink (already hit-incremented), its owning
3	 * Endpoint, and the incoming request, produce the Response the in-process
4	 * receiver should return — exercising a delivery failure mode on demand.
5	 *
6	 * All behaviors are implemented. The two with the most logic:
7	 * `fail-then-recover` (stateful — uses the Sink's hit count) and
8	 * `verify-signature` (recomputes the HMAC over `${timestamp}.${rawBody}`).
9	 */
10	
11	import { timingSafeEqual } from "node:crypto";
12	import { config } from "./config";
13	import { signBody } from "./delivery";
14	import type { Endpoint, Sink } from "./types";
15	
16	const sleep = (ms: number) => new Promise<void>((r) => setTimeout(r, ms));
17	
18	/**
19	 * Compute the Response for a single hit on `sink`.
20	 *
21	 * @param sink     the Sink, with `hits` already incremented for this request
22	 * @param endpoint the Endpoint that owns this Sink (carries the signing secret)
23	 * @param req      the inbound request (raw body + headers for signature checks)
24	 */
25	export async function behaviorResponse(
26		sink: Sink,
27		endpoint: Endpoint,
28		req: Request,
29	): Promise<Response> {
30		switch (sink.behavior) {
31			case "always-200":
32				return new Response("ok", { status: 200 });
33	
34			case "always-500":
35				return new Response("internal sink error", { status: 500 });
36	
37			case "410-gone":
38				// Exercises the classifier's permanent-failure + Endpoint-disable path.
39				return new Response("gone", { status: 410 });
40	
41			case "slow":
42				// Stalls long enough to trip the per-Attempt request timeout.
43				await sleep(config.sinkSlowDelayMs);
44				return new Response("slow ok", { status: 200 });
45	
46			case "fail-then-recover":
47				// Return 500 for the first N hits, then 200, so the worker
48				// retries and eventually succeeds. Use `sink.hits` (already incremented
49				// for this request) against a fail-count threshold.
50				// Fail-count threshold defined in config.failThenRecoverThreshold
51	
52				if (sink.hits <= config.failThenRecoverThreshold) {
53					return new Response("internal sink error", { status: 500 });
54				}
55				return new Response("ok", { status: 200 });
56	
57			case "verify-signature": {
58				const rawBody = await req.text();
59				const timestamp = req.headers.get("X-Webhook-Timestamp");
60				const signature = req.headers.get("X-Webhook-Signature");
61	
62				if (!timestamp || !signature) {
63					return new Response("Missing signature headers", { status: 401 });
64				}
65	
66				const signedPayload = `${timestamp}.${rawBody}`;
67				const expectedSignature = signBody(signedPayload, endpoint.secret);
68				const provided = Buffer.from(signature, "utf8");
69				const expected = Buffer.from(expectedSignature, "utf8");
70	
71				if (
72					provided.length !== expected.length ||
73					!timingSafeEqual(provided, expected)
74				) {
75					return new Response("Invalid signature", { status: 401 });
76				}
77	
78				return new Response("OK", { status: 200 });
79			}
80	
81			default: {
82				// Exhaustiveness guard: a new SinkBehavior must be handled above.
83				const _never: never = sink.behavior;
84				return new Response(`unknown behavior: ${_never}`, { status: 500 });
85			}
86		}
87	}

1	# 2. CLI + harness over the HTTP API; no web dashboard
2	
3	Date: 2026-06-17
4	
5	## Status
6	
7	Accepted
8	
9	## Context
10	
11	The assignment requires a way to register endpoints, ingest events, a test
12	harness, and observability. It explicitly allows the interface to be "an HTTP API,
13	a CLI harness, or some other interesting idea," and allows observability to be
14	"logs, metrics, status API, a dashboard, or something totally different."
15	
16	An initial plan included a Vite + shadcn/ui dashboard with tabs and action
17	buttons. Within a 2–6 hour budget, a polished frontend is the highest
18	effort-per-point item and the most likely to be left half-finished — which the
19	rubric penalizes more than a deliberately narrower scope.
20	
21	## Decision
22	
23	Cut the web dashboard. Build:
24	
25	- **One daemon** (`bun run dev`): Hono serving the ingest API, sink routes, the
26	  status/metrics API, and the in-process worker.
27	- **A CLI** that is a *thin client over the HTTP API* — never touching SQLite
28	  directly. Commands to add endpoints, create sinks, emit events, list
29	  deliveries/attempts, and show status.
30	- **A harness script** that drives the same HTTP API to run end-to-end scenarios
31	  and print a report.
32	- **Observability** = structured JSON logs from the daemon, a status/metrics JSON
33	  endpoint, and a `status --watch` CLI view.
34	
35	The CLI talks to the API, not the database, so there is a single writer (the
36	daemon owns the DB), and the CLI, harness, and any external consumer all exercise
37	the same surface.
38	
39	## Consequences

 * (delivered/failed/canceled) or the timeout elapses. Returns the final list.
 *
 * Throws on timeout so a scenario/test never silently asserts against a
 * still-in-flight queue. `expectedCount` (optional) also waits for at least that
 * many Deliveries to exist first, guarding the race where we poll before fan-out
 * rows are visible.
 */
export async function waitForSettled(
	baseUrl: string,
	opts: { timeout?: number; pollMs?: number; expectedCount?: number } = {},
): Promise<Delivery[]> {
	const timeout = opts.timeout ?? 5_000;
	const pollMs = opts.pollMs ?? 25;
	const deadline = Date.now() + timeout;

	for (;;) {
		const deliveries = await listDeliveries(baseUrl);
		const enough =
			opts.expectedCount === undefined ||
			deliveries.length >= opts.expectedCount;
		const allTerminal =
			deliveries.length > 0 && deliveries.every((d) => TERMINAL.has(d.status));

		if (enough && allTerminal) return deliveries;

		if (Date.now() > deadline) {
			const summary = deliveries
				.map((d) => `${d.id}=${d.status}`)
				.join(", ");
			throw new Error(
				`waitForSettled timed out after ${timeout}ms; ` +
					`deliveries (${deliveries.length}): [${summary}]`,
			);
		}
		await new Promise((r) => setTimeout(r, pollMs));
	}
}
