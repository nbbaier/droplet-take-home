34		const endpoint = await getEndpoint(delivery.endpointId);

38			endpoint.disabledAt ||

50			statusCode: result.statusCode,
51			responseBody: result.responseBody,
52			error: result.error,
53			durationMs: result.durationMs,
54		});
55	
56		const outcome = classifyResult(result);
57	
58		switch (outcome.action) {
59			case "delivered":
60				await markDelivered(delivery.id);
61				break;
62			case "failed":
63				await markFailed(delivery.id);
64				break;
65			case "gone":
66				await markFailed(delivery.id);
67				await disableEndpoint(delivery.endpointId);
68				await cancelDeliveriesForEndpoint(delivery.endpointId);
69				break;
70			case "retry":
71				if (shouldRetry(delivery.attemptCount, outcome)) {
72					const delayMs =
73						outcome.retryAfterMs ?? computeBackoffMs(delivery.attemptCount + 1);
74					const nextAttemptAt = new Date(Date.now() + delayMs).toISOString();
75					await rescheduleDelivery(delivery.id, nextAttemptAt);
76				} else {
77					await markFailed(delivery.id);
78				}
79				break;
80			default: {
81				const _exhaustive: never = outcome;
82				throw new Error(`Unknown outcome: ${JSON.stringify(_exhaustive)}`);
83			}
84		}
85	}

96		const deliveries = await claimBatch(config.claimBatchSize);
97		if (deliveries.length === 0) return;
98	
99		let index = 0;
100		const workerCount = Math.min(config.concurrency, deliveries.length);

102		await Promise.all(
103			Array.from({ length: workerCount }, async () => {
104				for (;;) {
105					const i = index++;
106					const delivery = deliveries[i];
107					if (!delivery) break;
108					await processDelivery(delivery);
109				}
110			}),
111		);
112	}

22		}

/** Pull `--url <value>` out of argv; fall back to the configured base URL. */

	const i = args.indexOf("--url");
