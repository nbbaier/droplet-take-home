const ANSI = {

function paint(code: string, s: string): string {
	return useColor ? `${code}${s}${ANSI.reset}` : s;

/** ISO → "YYYY-MM-DD HH:MM:SS". */
function shortTime(iso: string): string {
	return iso.replace("T", " ").slice(0, 19);

/** A "label   value" line, value right-aligned in a fixed column. */
function kv(label: string, value: number, labelWidth: number): string {
	return `  ${label.padEnd(labelWidth)}${String(value).padStart(5)}`;
}

/** Join two columns of pre-formatted (plain, un-colored) lines side by side. */
function sideBySide(left: string[], right: string[], gap = 4): string[] {
	const width = Math.max(...left.map((l) => l.length));
	const rows: string[] = [];
	for (let i = 0; i < Math.max(left.length, right.length); i++) {
		const l = left[i] ?? "";
		const r = right[i] ?? "";
		rows.push(l.padEnd(width + gap) + r);
	}
	return rows;
}

Now the harness-persistence issue. You've diagnosed it exactly: the harness spins up its own ephemeral, temp-DB daemon via startTestDaemon() and deletes it on exit, so nothing reaches webhooks.db and webhooks status (which queries the real daemon) sees nothing.

The clean fix is model A: make the harness a thin client of the real running daemon — which is exactly what ADR 0002 says the harness/CLI should be. Tests keep startTestDaemon (isolated temp). Concretely:

  • harness/index.ts: drop startTestDaemon; resolve a base URL (config.publicBaseUrl or --url), check the daemon is reachable, run the scenario against it. No teardown (not ours).
  • Scenarios: the unscoped ones (happy-path, retry-recovery, permanent-failure, signature-verified) currently waitForSettled({expectedCount:1}) then print all settled deliveries — fine against a fresh temp DB, but against the persistent accumulating webhooks.db they'd print the entire history. So they need the same endpointId scoping the tests and the other two scenarios already use.

The one workflow consequence: the harness stops being self-contained — you'd run bun run dev first (in another terminal), then bun run harness <scenario> and webhooks status both hit that daemon and share webhooks.db. For snappy demos you'd start dev with demo-fast timing, e.g. BACKOFF_BASE_MS=200 bun run dev.

That's the tradeoff: self-contained harness → "needs a daemon running," in exchange for real persistence + a meaningful status. I think it's the right call (and the more realistic story), but since it changes how you run things, want me to:

(a) go with model A (thin client, needs bun run dev), or (b) keep the harness self-contained but point its daemon at the real webhooks.db (persists data, but status still needs a separate bun run dev afterward to read it — so you get persistence without the clean "one daemon" story)?

I recommend (a). Say the word and I'll implement it (harness rewrite + scenario scoping), then verify the full loop: dev up → run a scenario → webhooks status shows the data.

import { config } from "../config";
import { type Scenario, scenarioNames, scenarios } from "./scenarios";

/** `--url <value>` from argv, else the configured base URL. */
function resolveBaseUrl(argv: string[]): string {
	const i = argv.indexOf("--url");
	if (i !== -1 && argv[i + 1]) return argv[i + 1] as string;
	return config.publicBaseUrl;
}

/** Positional args (everything that isn't `--url <value>`). */
function positionals(argv: string[]): string[] {
	const out: string[] = [];
	for (let i = 0; i < argv.length; i++) {
		if (argv[i] === "--url") {
			i++; // skip the value
			continue;
		}
		out.push(argv[i] as string);
	}
	return out;
}

/** Fail early with a helpful message if the daemon isn't up. */
async function assertDaemonReachable(baseUrl: string): Promise<void> {
	try {
		const res = await fetch(`${baseUrl}/`);
		if (!res.ok) throw new Error(`HTTP ${res.status}`);
	} catch (err) {
		const reason = err instanceof Error ? err.message : String(err);
		throw new Error(
			`Cannot reach the daemon at ${baseUrl} (${reason}).\n` +
				`Start it in another terminal first:\n` +
				`  bun run dev        # or: bun run demo   (snappy retries for a live demo)\n` +
				`then re-run the harness (or pass --url <baseUrl>).`,
		);
	}
}

	name: string,

### Retry policy

- Success: `2xx`. Retry: `5xx`, `429`, `408`, timeouts, connection errors.
- Permanent fail: other `4xx`. `410 Gone` → fail **and disable the endpoint**.
- Honor `Retry-After` on `429`/`503` (capped at the backoff cap).
- Exponential backoff + jitter, **5 attempts**, then terminal `failed`.

### Signature contract (for consumers verifying webhooks)

- Header `X-Webhook-Signature: sha256=<hex>` = HMAC-SHA256 over
  `` `${X-Webhook-Timestamp}.${rawBody}` `` keyed by the endpoint's secret.
- To verify: recompute over the timestamp + the exact received body — **not** the
  body alone. Timestamp in the signed payload gives replay protection.

import { createHmac, timingSafeEqual } from "node:crypto";

function verify(rawBody: string, headers: Headers, secret: string): boolean {
  const ts = headers.get("X-Webhook-Timestamp") ?? "";
  const got = headers.get("X-Webhook-Signature") ?? "";
  const want = `sha256=${createHmac("sha256", secret).update(`${ts}.${rawBody}`).digest("hex")}`;
  const a = Buffer.from(got);
  const b = Buffer.from(want);
  return a.length === b.length && timingSafeEqual(a, b);
}
